\section{scanf()}

\IFRU{Теперь попробуем использовать scanf().}{Now let's use scanf().}

\begin{lstlisting}
int main() 
{
	int x;
	printf ("Enter X:\n");

	scanf ("%d", &x);

	printf ("You entered %d...\n", x);

	return 0;
};
\end{lstlisting}
